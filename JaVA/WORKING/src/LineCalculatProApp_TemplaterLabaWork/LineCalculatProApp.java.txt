import wLineCalculat.wLineCalculat;

import java.io.File;
import java.io.FileWriter;
import java.io.IOException;

public class LineCalculatProApp {

    public static void main(String[] args) {

        File inf=new File("data.txt");

        try {
            inf.createNewFile();
            FileWriter writer=new FileWriter(inf);
            float a=2.7f;
            float b=2.7f;
            float c=2.7f;
            String text= String.valueOf(a);
            writer.write(text+"\n");
            text= String.valueOf(b);
            writer.write(text+"\n");
            text= String.valueOf(c);
            writer.write(text);
            writer.flush();
            writer.close();
        } catch (IOException e) {
            e.printStackTrace();
        }

        inf=new File("fraza.txt");

        try {
            inf.createNewFile();
            FileWriter writer=new FileWriter(inf);
            String text= "2.7 2.7 2.7";
            writer.write(text);
            writer.flush();
            writer.close();
        } catch (IOException e) {
            e.printStackTrace();
        }

        new wLineCalculat();
    }

}
