    public void paint(Graphics g) {
        Dimension rc = getSize();
        Insets in = getInsets();

        int RH = rc.width;
        int RW = rc.height;

        int minX = in.left;
        int maxX = RH - in.right;
        int minY = in.top;
        int maxY = RW - in.bottom;

        int cx, cy, dx, dy;

        int x1, y1, x2, y2, x3, y3;

        Graphics2D g2 = (Graphics2D) g;
        g2.setStroke(new BasicStroke(1.0f));

        cx = Math.abs(maxX - minX) / 2;
        cy = Math.abs(maxY - minY) / 2 + 50;
        dx = 40;
        dy = 20;

        // TODO THIS PAINT
        x=15; y=65; dy=dx=30;
        String text = "";
        String ms = "Тестові завдання:";
        g.setFont(new Font("Trebuchet Ms", Font.BOLD, 24));
        g.setColor(new Color(35, 67, 221));
        g.drawString(ms, x, y);
        File inf = new File("data1.txt");
        BufferedReader reader = null;
        try {
            Scanner scan = new Scanner(inf);
            int i = 1;
            while (scan.hasNextLine()) {
                text = scan.nextLine();
                g.setColor(new Color(135, 18, 153));
                g.drawString(text, x, y + i * dy);
                int sum=0;
                for (int k = 0; k <text.length() ; k++) {
                    if(text.charAt(k)=='+' || text.charAt(k)=='-' || text.charAt(k)=='*')
                        sum++;
                }
                String res=" усього: "+sum;
                g.setColor(new Color(213, 0, 183));
                g.drawString(res, x + text.length()*dx/2, y + i * dy);
                i++;
            }
            scan.close();
        } catch (FileNotFoundException e) {
            e.printStackTrace();
        }
    }
