	cout << "\n\tПрактическая работа. ТЕМА: Заполнение квадратной матрицы. \n\n";

	int n,k,i,j,p;
	int A[50][50];
	
	printf("\n\tВведите размерность матрицы N=");
	cin>>n;
	cout << "\n\t";

	i = n*n;
	p = n / 2;
	for (k = 1; k <= p; k++)/*Цикл по номеру витка*/
	{
		/*Определение значений верхнего гор столбца*/
		for (j = k - 1; j<n - k + 1; j++) A[k - 1][j] = i--;
		/* --//-- По правому вертикальному столбцу*/
		for (j = k; j<n - k + 1; j++) A[j][n - k] = i--;
		/* --//-- по нижнему горизонтальному столбцу*/
		for (j = n - k - 1; j >= k - 1; --j) A[n - k][j] = i--;
		/* --//-- по левому вертикальному столбцу*/
		for (j = n - k - 1; j >= k; j--) A[j][k - 1] = i--;
	}
	if (n % 2 == 1) A[p][p] = 1;

	printf("\n\t");
	for (i = 0; i < n; i++)
	{

		for (j = 0; j < n; j++)
		{
			printf("%5i\t", A[i][j]);
		}
		printf("\n\t");
	}

#include <iostream>
#include <time.h>

using namespace std;

int main()
{
	srand((unsigned)time(NULL));
	setlocale(LC_ALL, "rus");
	cout << "\n\tПрактическая работа. ТЕМА: Обработка линейных таблиц.\n\n";

	int n, k;
	double TT[50];
	double aa, bb, sum;
	sum = 0;

	cout << "\tВведите число (5<N<30) N=";
	cin >> n;

	aa = -99.0;
	bb = 99.0;

	sum = 0;

	printf("\n\n\tИсходная матрица чисел:\n\n");
	for (k = 0; k < n; k++)
	{
		TT[k] = aa + (double)rand() / RAND_MAX*(bb - aa);
		printf("%8.2f", TT[k]);
		
		if (TT[k] > 0)
		{
			sum += powf(TT[k], 2.0);
		}

	}

	printf("\n\n\tСумма квадратов положительных чисел равна:%8.2f", sum);

	cout << endl << endl << "\t";
	system("pause");
	return 0;
}
