#include <iostream>

// додаткова бібліотека 
#include <time.h>

#define roz 8

using namespace std;

int main()
{
	// генератор випадкових чисел
	srand((unsigned)time(NULL));
	setlocale(LC_ALL, ".1251");
	system("color 1F");

	// розміри матрицi
	int A[roz], B[2*roz], C[2*roz];
	int n, k, i, p, sum;

	n = roz;
	int a = 0;
	int b = 10;


	system("cls");
	printf("\n\tЗаповнення матрицi A[%i] цiлими числами. \n\n", n);

	for (k = 0; k < n; k++)
	{
		A[k] = a + rand() % (b - a + 1);
		printf("\tA[%i]=%i", k, A[k]);
	}

	cout << "\n\n\tBведiть число для пошуку M=";
	int m;
	cin >> m;

	for (i = 0; i < n; i++)
	{
		B[i] = -1;
		C[i] = -1;
	}
	p = 0;

	for (k = 0; k < n - 1; k++)
	{
		for (i = k+1; i < n; i++)
		{
			sum = A[k] + A[i];
			if (sum == m)
			{
				B[p] = A[k];
				C[p] = A[i];
				p++;
			}
		}
	}
	cout << "\n\n\tРезультат пошуку:\n";
	for (k = 0; k < roz+roz/2; k++)
	{
	    if(B[k]+C[k]==m)
		printf("\n\t%i+%i=%i",  B[k], C[k], m);
	}


	cout << endl << endl << "\t";
	system("pause");
	return 0;
}
